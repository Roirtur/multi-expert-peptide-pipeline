\section{limitations et défis rencontrés}
\begin{frame}
    \frametitle{Limitations et défis rencontrés}
    \begin{itemize}
        \item Manque de données biologiques pour entraîner le modèle du biologiste
        \item Complexité de la modélisation des interactions biologiques
        \item Difficulté à évaluer la performance du pipeline sans tests biologiques
        \item Besoin d'une validation expérimentale pour confirmer les résultats obtenus \\
    \end{itemize}
\end{frame}
