\subsection{Expert Chimiste}
\begin{frame}
    \frametitle{Chimiste}
    \begin{block}{Modèle du chimiste}
        \begin{itemize}
        \item Expert évaluant les propriétés chimiques des peptides générés
        \item Basé sur des règles chimiques et des calculs mathématiques pour évaluer la validité et les propriétés des peptides
        \item Capable de filtrer les peptides générés en fonction de critères chimiques (ex: solubilité, stabilité)
        \end{itemize}
    \end{block}
    \begin{block}{Propriétés chimiques évaluées}
        \begin{itemize}
            \item Taille de la séquence
            \item Poids moléculaire
            \item Charge nette
            \item LogP (coefficient de partage octanol-eau)
            \item Hydrophobicité
            \item Point isoélectrique
        \end{itemize}
    \end{block}
\end{frame}
